<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" styleName="plain" viewSourceURL="srcview/index.html">
  
  <mx:Script>
  <![CDATA[
  private var testShader:Shader;
  
  private function init():void
  {
    [Embed(source="VerySimpleFilter.pbj", mimeType="application/octet-stream")]
    var TestShaderClass:Class;
    
    testShader = new Shader(new TestShaderClass());
    
    testShader.data.exposure.value = [0.5];
    
    var bitmapData:BitmapData = new BitmapData(i.width, i.height);
    bitmapData.draw(i, new Matrix());
    testShader.data.src.input = bitmapData;
    
    //var shaderFilter:ShaderFilter = new ShaderFilter(testShader);
    //i.filters = [shaderFilter];
    
    var shaderJob:ShaderJob = new ShaderJob(testShader);
    var jobResult:BitmapData = new BitmapData(i.width, i.height);
    shaderJob.target = jobResult;
    shaderJob.start(true);
    
    uic.addChild(new Bitmap(jobResult));
    uic.width = i.width;
    uic.height = i.height;
  }
  ]]>
  </mx:Script>

  <mx:Button id="b" label="loading image" click="init()" enabled="false"/>
  
  <mx:Image id="i" source="test.jpg" complete="b.label = 'run filter'; b.enabled = true;"/>
  
  <mx:UIComponent id="uic"/>
  
</mx:Application>
