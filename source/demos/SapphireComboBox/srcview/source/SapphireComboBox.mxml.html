<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" verticalAlign="middle" viewSourceURL="srcview/index.html">
  
  <mx:Script>
    import mx.collections.ArrayCollection;
    import mx.managers.PopUpManager;
    import mx.events.ItemClickEvent;
    import mx.core.IFlexDisplayObject;
  </mx:Script>
  
  <mx:Button id="b" label="Year" cornerRadius="0">
    <mx:mouseOver>
    <![CDATA[
      var s:PopUpScroller = new PopUpScroller();
      
      var years:Array = new Array();
      
      for (var i:int = 1900; i < 2010; i++)
      {
        years.push(i.toString());
      }
      
      s.dataProvider = new ArrayCollection(years);
    
      s.width = b.width + 10;
      s.x = b.x - 5;
      s.y = b.y - 150;
      
      s.addEventListener("change", function(event:ItemClickEvent):void {
        b.label = event.item as String;
        PopUpManager.removePopUp(IFlexDisplayObject(event.currentTarget));
      }, false, 0, true);
    
      PopUpManager.addPopUp(s, this);
    ]]>
    </mx:mouseOver>
  </mx:Button>
  
</mx:Application>
