<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas xmlns:mx="http://www.adobe.com/2006/mxml" height="300" backgroundColor="#ffffff" verticalScrollPolicy="off">
  
  <mx:Metadata>
    [Event(name="change", type="mx.events.ItemClickEvent")]
  </mx:Metadata>
  
  <mx:Script>
    import mx.managers.PopUpManager;
    import mx.events.ItemClickEvent;
  </mx:Script>
  
  <mx:ArrayCollection id="dataProvider"/>
  
  <mx:Boolean id="scrollUp">false</mx:Boolean>
  <mx:Boolean id="scrollDown">false</mx:Boolean>
  
  <mx:creationComplete>
  <![CDATA[
    var t:Timer = new Timer(10);
    t.addEventListener(TimerEvent.TIMER, function(event:TimerEvent):void {
      if ((scrollUp) && (c.y < -9))
      {
        c.y += 10;
      }
      if ((scrollDown) && (c.y > ((c.height * -1) + 300)))
      {
        c.y -= 10;
      }
    });
    t.start();
  ]]>
  </mx:creationComplete>
  
  <mx:mouseOut>
  <![CDATA[
    // don't close this popup unless the mouse really left
    if ((event.relatedObject != null) && (!this.contains(event.relatedObject)) && (event.relatedObject != this)) 
    {
      PopUpManager.removePopUp(this);
    }
  ]]>
  </mx:mouseOut>
  
  <mx:mouseMove>
  <![CDATA[
    if (event.target == this)
    {
      if ((event as MouseEvent).localY < 50)
      {
        scrollUp = true;
      }
      else
      {
        scrollUp = false;
      }
      
      if ((event as MouseEvent).localY > 250)
      {
        scrollDown = true;
      }
      else
      {
        scrollDown = false;
      }
    }
  ]]>
  </mx:mouseMove>
  
  <mx:VBox id="c" width="100%" verticalScrollPolicy="off">
    <mx:Repeater id="rp" dataProvider="{dataProvider}">
      <mx:Label text="{rp.currentItem}" width="100%" fontSize="16" textAlign="center"
        mouseOver="event.currentTarget.setStyle('fontWeight', 'bold')"
        mouseOut="event.currentTarget.setStyle('fontWeight', 'normal')"
        mouseDown="dispatchEvent(new ItemClickEvent('change', false, false, event.currentTarget.getRepeaterItem(), dataProvider.getItemIndex(event.currentTarget.getRepeaterItem()), null, event.currentTarget.getRepeaterItem()))"
        />
    </mx:Repeater>
  </mx:VBox>
  
</mx:Canvas>
