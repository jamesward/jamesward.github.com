<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" viewSourceURL="srcview/index.html">

  <mx:Script>
  <![CDATA[
    import mx.messaging.messages.RemotingMessage;
    import mx.utils.UIDUtil;
    import mx.rpc.events.ResultEvent;
  
    import flash.utils.getTimer;
  
    private const testUrl:String = "http://www.jamesward.com/census/messagebroker/amf?id=flex_amf3&gzip=false";
  
    private var startTime:int;
    
    private var numTestsRun:int = 0;
    
    private var nc:NetConnection = new NetConnection();
    
    private var loader:URLLoader;
    
    private function startTest():void
    {
      pb.indeterminate = true;
      
      resetTimer();
      
      startRemoteObjectTest();
    }
    
    private function resetTimer():void
    {
      numTestsRun = 0;
      startTime = flash.utils.getTimer();
    }
    
    private function startRemoteObjectTest():void
    {
      ro1.endpoint = getUIDTestUrl();
      
      ro1.getElements(0, 20000);
    }
    
    private function startNetConnectionTest():void
    {
          nc.connect(getUIDTestUrl());
 
            var responder:Responder = new Responder(handleNetConnectionResult, onFault);
 
            nc.call("census-ro.getElements", responder, 0, 20000);
        }
        
        private function startURLLoaderTest():void
    {
      ro2.channelSet.channels[0].uri = getUIDTestUrl();
      
      ro2.getElements(0, 20000);
        }
    
    private function handleRemoteObjectResult(event:ResultEvent):void
    {
      numTestsRun++;
      
      if (numTestsRun < s.value)
      {
        startRemoteObjectTest();
        
        return;
      }
      
      ta.text += "RemoteObject " + (flash.utils.getTimer() - startTime) + "\n";
      
      
      resetTimer();

      startNetConnectionTest();
    }
    
    private function handleNetConnectionResult(result:*):void
    {
      numTestsRun++;
      
      if (numTestsRun < s.value)
      {
        startNetConnectionTest();
        
        return;
      }
      
      ta.text += "NetConnection " + (flash.utils.getTimer() - startTime) + "\n";
      
      
      resetTimer();
      
      startURLLoaderTest();
    }
    
    private function handleURLLoaderResult(event:ResultEvent):void
    {
      numTestsRun++;
      
      if (numTestsRun < s.value)
      {
        startURLLoaderTest();
        
        return;
      }
      
      ta.text += "URLLoader " + (flash.utils.getTimer() - startTime) + "\n";


      pb.indeterminate = false;
    }
    
    private function onFault(e:*):void
        {
            trace(e);
        }
        
        private function getUIDTestUrl():String
        {
          return testUrl + "&uid=" + UIDUtil.createUID();
        }
    ]]>
  </mx:Script>

  <mx:RemoteObject id="ro1" destination="census-ro" result="handleRemoteObjectResult(event)"/>
  
  <mx:RemoteObject id="ro2" destination="census-ro" result="handleURLLoaderResult(event)">

      </mx:ChannelSet>
    </mx:channelSet>
  </mx:RemoteObject>

  <mx:ProgressBar id="pb" label="" width="100%" labelPlacement="center"/>
  
  
  <mx:FormItem label="Number of iterations:">
    <mx:HSlider id="s" minimum="1" maximum="10" allowTrackClick="true"
      thumbCount="1" tickInterval="1" snapInterval="1" dataTipPrecision="0" labels="[1, 10]"/>
  </mx:FormItem>
  
  <mx:ControlBar>
    <mx:Button label="start test" click="startTest()"/>
  </mx:ControlBar>
  
  </mx:Panel>
